CONST max = 10;
VAR i, sum;
PROCEDURE calculateSum ;
  VAR temp ;
  BEGIN
    temp := 0 ;
    i := 1 ;
    WHILE i <= max DO
      BEGIN
        temp := temp + i ;
        i := i + 1;
      END ;
    sum := temp
  END;
BEGIN
  CALL calculateSum ;
  IF sum = 55 THEN
    sum := 100 ;
  WHILE sum > 0 DO
    sum := sum - 1
END.
